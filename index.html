const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); canvas.width = 600; canvas.height = 800; const poems = [ { title: "靜夜思", content: "床前明月光疑是地上霜舉頭望明月低頭思故鄉" }, { title: "將進酒", content: "君不見黃河之水天上來奔流到海不復回君不見高堂明鏡悲白髮朝如青絲暮成雪" }, { title: "蜀道難", content: "噫吁嚱危乎高哉蜀道之難難於上青天蠶叢及魚鳧開國何茫然爾來四萬八千歲不與秦塞通人煙" } ]; const commonWords = "的一是在不了有和人這中大來上個國我到說們為子和社多時樣後自下以生可能而面分於主行過家開現前出民日發正道成前那西對也此法之實重物度法後政也"; let gameState = 'menu'; let score = 0; let currentIndex = 0; let currentPoemData = null; let entities = []; let particles = []; let player = { x: 300, y: 750, width: 40, height: 40 }; let keys = {}; let lastSpawnTime = 0; let lastFireTime = 0; // 新增射擊冷卻計時 let gameSpeed = 1; window.addEventListener('keydown', e => keys[e.code] = true); window.addEventListener('keyup', e => keys[e.code] = false); function startGame(index) { currentPoemData = poems[index]; currentIndex = 0; score = 0; entities = []; particles = []; gameSpeed = 1 + (index * 0.5); gameState = 'playing'; document.getElementById('start-screen').style.display = 'none'; document.getElementById('current-poem').innerText = "正在背誦：《" + currentPoemData.title + "》"; updateTargetUI(); requestAnimationFrame(gameLoop); } function updateTargetUI() { const target = currentPoemData.content[currentIndex]; document.getElementById('target-word').innerText = target ? "下一個字：" + target : "已完成！"; document.getElementById('score').innerText = score; } function spawnWord() { const now = Date.now(); if (now - lastSpawnTime > 1200 / gameSpeed) { // 提高正確字出現機率，確保遊戲流暢 const shouldSpawnCorrect = Math.random() > 0.5; let char = ''; if (shouldSpawnCorrect && currentIndex < currentPoemData.content.length) { char = currentPoemData.content[currentIndex]; } else { char = commonWords[Math.floor(Math.random() * commonWords.length)]; } entities.push({ x: Math.random() * (canvas.width - 60) + 30, y: -30, text: char, isBullet: false, speed: (1.5 + Math.random() * 1.5) * gameSpeed }); lastSpawnTime = now; } } function gameLoop() { if (gameState !== 'playing') return; ctx.clearRect(0, 0, canvas.width, canvas.height); // 1. 玩家移動 if (keys['ArrowLeft'] && player.x > 0) player.x -= 7; if (keys['ArrowRight'] && player.x < canvas.width - player.width) player.x += 7; // 2. 射擊邏輯 (冷卻時間 250ms) const now = Date.now(); if (keys['Space'] && now - lastFireTime > 250) { entities.push({ x: player.x + 20, y: player.y, isBullet: true, speed: -10 }); lastFireTime = now; } // 3. 繪製玩家 ctx.fillStyle = '#3d405b'; ctx.beginPath(); ctx.moveTo(player.x + 20, player.y); ctx.lineTo(player.x, player.y + 40); ctx.lineTo(player.x + 40, player.y + 40); ctx.fill(); // 4. 更新與碰撞偵測 for (let i = entities.length - 1; i >= 0; i--) { let e = entities[i]; e.y += e.speed; if (e.isBullet) { ctx.fillStyle = 'black'; ctx.beginPath(); ctx.arc(e.x, e.y, 5, 0, Math.PI * 2); ctx.fill(); // 檢查碰撞 for (let j = entities.length - 1; j >= 0; j--) { let target = entities[j]; if (!target.isBullet && Math.abs(e.x - target.x) < 30 && Math.abs(e.y - target.y) < 30) { if (target.text === currentPoemData.content[currentIndex]) { score += 10; currentIndex++; gameSpeed += 0.03; createParticles(target.x, target.y, 'rgba(0,0,0,0.6)'); updateTargetUI(); if (currentIndex >= currentPoemData.content.length) endGame(); } else { score = Math.max(0, score - 5); document.getElementById('score').innerText = score; createParticles(target.x, target.y, 'rgba(200,0,0,0.6)'); } entities.splice(j, 1); // 移除文字 entities.splice(i > j ? i - 1 : i, 1); // 移除子彈 (注意索引變化) break; } } } else { ctx.fillStyle = '#3d405b'; ctx.font = "bold 32px 'STKaiti', 'Microsoft JhengHei'"; ctx.textAlign = "center"; ctx.fillText(e.text, e.x, e.y); } if (e && (e.y > canvas.height || e.y < -50)) entities.splice(i, 1); } // 5. 粒子效果 for (let i = particles.length - 1; i >= 0; i--) { let p = particles[i]; p.x += p.vx; p.y += p.vy; p.life -= 0.02; ctx.fillStyle = p.color.replace(/[\d.]+\)$/g, p.life + ')'); ctx.beginPath(); ctx.arc(p.x, p.y, 6 * p.life, 0, Math.PI * 2); ctx.fill(); if (p.life <= 0) particles.splice(i, 1); } spawnWord(); if (gameState === 'playing') requestAnimationFrame(gameLoop); } function createParticles(x, y, color) { for (let i = 0; i < 10; i++) { particles.push({ x, y, vx: (Math.random() - 0.5) * 6, vy: (Math.random() - 0.5) * 6, life: 1, color }); } } function endGame() { gameState = 'menu'; setTimeout(() => { alert("詩篇完成！總分：" + score); const hs = localStorage.getItem('poetHighScore') || 0; if (score > hs) localStorage.setItem('poetHighScore', score); document.getElementById('start-screen').style.display = 'flex'; updateHighScore(); }, 100); } function updateHighScore() { document.getElementById('high-score').innerText = localStorage.getItem('poetHighScore') || 0; }
